methods GET

Get all users

query string address: /users

not request 

response data: {
        resultCode: 0 | 1,
        messages: [string],
        data: [{
               id: string,
               name: string,
               wishlist: [string],
               shoppingList: [string],
        }]
}

Get user by ID

query string address: /user?id=string

request id:"string"
example: /user?id=61a6286353b5dad92e57b4c0

response data: {
        resultCode: 0 | 1,
        messages: [string],
        data: {
               id: string,
               name: string,
               wishlist: [string],
               shoppingList: [string],
        }
}

}

                cost: string,
                currency: string,

}

methods POST

Add user

query string address: /user

request data: {"name":"string"}
example:{"name":"Tester"}

response data: {
        resultCode: 0 | 1,
        messages: [string],
        data: {
            id: string,
            name: string,
            wishlist: [string],
            shoppingList: [string],
        }
}

Get products by list

query string address: /products

request data: {"listProductsId":[string]}
example:{"listProductsId":["61a61a5138f1fb7fb25fde75","61a61a5138f1fb7fb25fde73", "61a61eb74ef1f443c0cb654e"]}

response data: {
        resultCode: 0 | 1,
        messages: [string],
        data: [{
                type:string,
                span:number,
                data: [productData]
        }]
}

Change wishlist

query string address: /user/wishlist

request data: {"userId":"string", "productId":"string"}
example: {"userId":"61a6286353b5dad92e57b4c0", "productId":"61a61eb74ef1f443c0cb654e"}

response data: {
        resultCode: 0 | 1,
        messages: [string],
        data: {
                id: string,
                name: string,
                wishlist: [string],
                shoppingList: [string],
         }
}

Change purchase

query string address: /user/purchase

request data: {"userId":"string","productId":"string" , isAdd:"boolean"}
isAdd: boolean - if this field is true, the product is added to the purchase list, if this field is false, the product is removed from the purchase list
example:{"userId":"61a6286353b5dad92e57b4c0", "productId":"61a61eb74ef1f443c0cb654e", isAdd:true}

response data: {
        resultCode: 0 | 1,
        messages: [string],
        data: {
                id: string,
                name: string,
                wishlist: [string],
                shoppingList: [string],
         }
}
